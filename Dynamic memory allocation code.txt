#include <stdio.h>
#include <time.h>

void firstFit(int blockSize[], int blocks, int processSize[], int processes) {
    int allocation[10];

    for (int i = 0; i < processes; i++)
        allocation[i] = -1;

    int temp[10];
    for (int i = 0; i < blocks; i++)
        temp[i] = blockSize[i];

    for (int i = 0; i < processes; i++) {
        for (int j = 0; j < blocks; j++) {
            if (temp[j] >= processSize[i]) {
                allocation[i] = j;
                temp[j] -= processSize[i];
                break;
            }
        }
    }

    printf("\nFirst Fit\n");
    printf("Process\tSize\tPartition No\n");

    for (int i = 0; i < processes; i++) {
        if (allocation[i] != -1)
            printf("P%d\t%d\t%d\n", i, processSize[i], allocation[i] + 1);
        else
            printf("P%d\t%d\tNot Allocated\n", i, processSize[i]);
    }

    printf("Free Partitions:\n");
    for (int i = 0; i < blocks; i++)
        printf("Partition %d = %d\n", i + 1, temp[i]);
}

void bestFit(int blockSize[], int blocks, int processSize[], int processes) {
    int allocation[10];

    for (int i = 0; i < processes; i++)
        allocation[i] = -1;

    int temp[10];
    for (int i = 0; i < blocks; i++)
        temp[i] = blockSize[i];

    for (int i = 0; i < processes; i++) {
        int best = -1, diff = 10000;

        for (int j = 0; j < blocks; j++) {
            if (temp[j] >= processSize[i]) {
                if (best == -1 || temp[j] - processSize[i] < diff) {
                    best = j;
                    diff = temp[j] - processSize[i];

                    if (diff == 0)
                        break;
                }
            }
        }

        if (best != -1) {
            allocation[i] = best;
            temp[best] -= processSize[i];
        }
    }

    printf("\nBest Fit\n");
    printf("Process\tSize\tPartition No\n");

    for (int i = 0; i < processes; i++) {
        if (allocation[i] != -1)
            printf("P%d\t%d\t%d\n", i, processSize[i], allocation[i] + 1);
        else
            printf("P%d\t%d\tNot Allocated\n", i, processSize[i]);
    }

    printf("Free Partitions:\n");
    for (int i = 0; i < blocks; i++)
        printf("Partition %d = %d\n", i + 1, temp[i]);
}

void worstFit(int blockSize[], int blocks, int processSize[], int processes) {
    int allocation[10];

    for (int i = 0; i < processes; i++)
        allocation[i] = -1;

    int temp[10];
    for (int i = 0; i < blocks; i++)
        temp[i] = blockSize[i];

    for (int i = 0; i < processes; i++) {
        int worst = -1;

        for (int j = 0; j < blocks; j++) {
            if (temp[j] >= processSize[i]) {
                if (worst == -1 || temp[j] > temp[worst])
                    worst = j;
            }
        }

        if (worst != -1) {
            allocation[i] = worst;
            temp[worst] -= processSize[i];
        }
    }

    printf("\nWorst Fit\n");
    printf("Process\tSize\tPartition No\n");

    for (int i = 0; i < processes; i++) {
        if (allocation[i] != -1)
            printf("P%d\t%d\t%d\n", i, processSize[i], allocation[i] + 1);
        else
            printf("P%d\t%d\tNot Allocated\n", i, processSize[i]);
    }

    printf("Free Partitions:\n");
    for (int i = 0; i < blocks; i++)
        printf("Partition %d = %d\n", i + 1, temp[i]);
}

int main() {
    int blocks, processes;

    printf("Enter number of memory partitions: ");
    scanf("%d", &blocks);

    int blockSize[10];

    printf("Enter partition sizes:\n");
    for (int i = 0; i < blocks; i++)
        scanf("%d", &blockSize[i]);

    printf("Enter number of processes: ");
    scanf("%d", &processes);

    int processSize[10];

    printf("Enter process sizes:\n");
    for (int i = 0; i < processes; i++)
        scanf("%d", &processSize[i]);

    clock_t start, end;
    double time_taken;

    start = clock();
    firstFit(blockSize, blocks, processSize, processes);
    end = clock();
    time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("Execution Time (First Fit): %f seconds\n", time_taken);

    start = clock();
    bestFit(blockSize, blocks, processSize, processes);
    end = clock();
    time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("Execution Time (Best Fit): %f seconds\n", time_taken);

    start = clock();
    worstFit(blockSize, blocks, processSize, processes);
    end = clock();
    time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("Execution Time (Worst Fit): %f seconds\n", time_taken);

    printf("\nAnalysis:\n");
    printf("First Fit : Fast allocation, moderate fragmentation.\n");
    printf("Best Fit : Slower, minimum leftover space but more small fragments.\n");
    printf("Worst Fit : Slowest, leaves large free blocks, may waste memory.\n");
    printf("Deallocation: Freed partitions can be reused dynamically.\n");
    printf("External Fragmentation occurs in all dynamic partition methods.\n");

    return 0;
}
